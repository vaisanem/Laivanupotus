14  public class ControlPanel implements Runnable {
15      
16      private JFrame frame;
17      private Game game;
18      private JButton jenson;
19      private JButton jenson2;
20  
21      public ControlPanel(Game game) {
22          this.game = game;
23      }
24  
25      @Override
26      public void run() {
27          frame = new JFrame("Battleship");
28  
29          frame.setPreferredSize(new Dimension(400, 600));
30  
31          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
32  
33          createComponents(frame.getContentPane());
34  
35          frame.pack();
36          frame.setVisible(true);
37      }
38      
39      public void createComponents(Container container) {
40          container.setLayout(new GridLayout(12, 1));
41          JLabel label = new JLabel("PLACE YOUR SHIP");
42          label.setHorizontalAlignment(JLabel.CENTER);
43          JLabel message = new JLabel("");
44          JLabel info = new JLabel("Details in order (int): length, x, y, direction");
45          JTextField text = new JTextField();
46          text.setText(" length, x, y, direction 1 (Up), 2 (Right), 3 (Down), 4 (Left)");
47          jenson = new JButton("Enter");
48          jenson.addActionListener(new ShipListener(game, jenson, text, message));
49          JLabel space = new JLabel("_________________________________________________________________");
50          JLabel label2 = new JLabel("SHOOT A MISSILE");
51          label2.setHorizontalAlignment(JLabel.CENTER);
52          JLabel message2 = new JLabel("");
53          JLabel info2 = new JLabel("Details in order (int): x, y");
54          JTextField text2 = new JTextField();
55          text2.setText("x, y");
56          jenson2 = new JButton("Enter");
57          jenson2.addActionListener(new MissileListener(game, jenson2, text2, message2));
58          container.add(label);
59          container.add(info);
60          container.add(message);
61          container.add(text);
62          container.add(jenson);
63          container.add(space);
64          container.add(label2);
65          container.add(info2);
66          container.add(message2);
67          container.add(text2);
68          container.add(jenson2);
69      }
70      
71      public void disableShipButton() {
72          jenson.setEnabled(false);
73      }
74      
75      public void disableMissileButton() {
76          jenson2.setEnabled(false);
77      }
78      
79      public void enableMissileButton() {
80          jenson2.setEnabled(true);
81      }
82  }
